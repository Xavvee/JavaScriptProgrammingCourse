function calculate_dot(A, B)
    if (size(A, 1) != 1 || size(B, 1) != 1)
        print("Niepoprawne wymiary tablicy")
        return 0
    end
    return dot(A, B)
end

###Mnożenie przez wektor

function calculate_multi(A, B)
    if (size(B, 2) != 1)
        print("B powinno być wektorem!")
        return
    end
    if (size(A, 2) != size(A, 1))
        print("Macierz A nie jest kwadratowa!")
        return
    end
    if (size(A, 2) != size(B, 1))
        print("Mnożenie niemożliwe z powodu niezgodności ilości wierszy i kolumn!")
        return
    end
    return *(A, B)
end

####Pomiary dla iloczynu skalarnego
function measure_scalar_time(times)
    calculate_dot([1],[2])
    timings=zeros(times,12)
    size=20000
    for i in 1:times
        A=rand(1,size)
        B=rand(1,size)
        for j in 1:10
            println("Skalar: "*string(i)*string(j))
            timings[i,j]=@elapsed calculate_dot(A,B)
        end
        timings[i,11]=size
        timings[i,12]=0
        size+=30000
    end
    return timings
end


####Pomiary dla mnożenia przez wektor
function measure_multi_time(times)
    calculate_multi([1 1 1 1; 2 2 2 1; 3 3 3 3; 4 4 4 4],[1; 2; 3; 4])    
    timings=zeros(times,12)
    size=5000
    for i in 1:times
        A=rand(size,size)
        B=rand(size)
        for j in 1:10
            println("Multi: "*string(i)*string(j))
            timings[i,j]=@elapsed calculate_multi(A,B)
        end
        timings[i,11]=size
        timings[i,12]=1
        size+=1400
    end
    return timings
end

using LinearAlgebra
using DelimitedFiles
using Random
using CSV, Tables
using DataFrames

function combine_output(multi, scalar, times)
    timings=zeros(2*times, 12)
    for i in 1:times
        for j in 1:12
            timings[i, j] = scalar[i, j]
        end
    end   
    for i in (times+1):times*2
        for j in 1:12
            timings[i, j] = multi[i-times, j]
        end
    end
    return timings
end

function save_to_csv(filename::AbstractString, data::Array{Float64, 2})
    headers = [["Time $i" for i in 1:(size(data, 2)-2)]...]
    headers = [headers..., "Size", "Type"]
    tab = Tables.table(data)
    CSV.write(filename, DataFrame(tab), header=headers, delim=";")
end

using DataFrames
using CSV
using Plots
using Statistics

input="output.csv"
mydata=CSV.read(input, delim=";",DataFrame)

means_scalar = [mean(mydata[i, 1:10]) for i in 1:div(size(mydata, 1), 2)]
means_multi = [mean(mydata[i, 1:10]) for i in (div(size(mydata, 1), 2)+1):size(mydata, 1)]

std_scalar = [std(mydata[i, 1:10]) for i in 1:div(size(mydata, 1), 2)]
std_multi=[std(mydata[i, 1:10]) for i in (div(size(mydata, 1), 2)+1):size(mydata, 1)]

###Przygotowanie danych dla wykresów 
n = div(size(mydata, 1), 2)
scalar_size = mydata[:, 11][1:n]
multi_size=mydata[:,11][n+1:2*n]

###Zdefiniowanie wykresów
p1 = plot(scalar_size, means_scalar, yerror=std_scalar, seriestype=:scatter, xlabel="Rozmiar", ylabel="Średni czas[s]", title="Średnie i odchylenia standardowe dla mnożenia skalarnego")
p2 = plot(multi_size, means_multi, yerror=std_multi, seriestype=:scatter, xlabel="Rozmiar", ylabel="Średni czas[s]", title="Średnie i odchylenia standardowe dla mnożenia przez wektor")

###Połączenie wykresów za pomocą layouts
plot(p1, p2, layout=(2,1), legend=false, size=(1200,1200))

Z otrzymanych wykresów można wywnioskować, że mnożenie przez wektor zajmuje więcej czasu oraz gwałtownie zwiększa swój czas działania, podczas gdy iloczyn skalarny uzyskuje podobne czasy niezależnie od wielkości wektorów.
